045
046public class TestPersistenceXmlLoading {
047
048  public TestPersistenceXmlLoading() {
049    super();
050  }
051
052  @Test
053  public void testLoadingXml() throws JAXBException {
054    final JAXBContext jaxbContext = JAXBContext.newInstance("org.microbean.jpa.jaxb");
055    assertNotNull(jaxbContext);
056    final Unmarshaller unmarshaller = jaxbContext.createUnmarshaller();
057    unmarshaller.setEventHandler(new javax.xml.bind.helpers.DefaultValidationEventHandler());
058    final Persistence p = (Persistence)unmarshaller.unmarshal(Thread.currentThread().getContextClassLoader().getResource(this.getClass().getSimpleName() + "/persistence.xml"));
059    assertNotNull(p);
060    final Collection<? extends PersistenceUnit> persistenceUnits = p.getPersistenceUnit();
061    assertNotNull(persistenceUnits);
062    assertEquals(1, persistenceUnits.size());
063    final PersistenceUnit persistenceUnit = persistenceUnits.iterator().next();
064    assertNotNull(persistenceUnit);
065    final Boolean excludeUnlistedClasses = persistenceUnit.isExcludeUnlistedClasses();
066    assertNull(excludeUnlistedClasses);
067  }
068
069  @Test
070  public void testLoadingYaml() throws IOException {
071    final URL url = Thread.currentThread().getContextClassLoader().getResource(this.getClass().getSimpleName() + "/persistence.yaml");
072    final Yaml yaml = new Yaml();
073    try (final InputStream inputStream = url.openStream()) {
074      final Map<String, Object> data = yaml.load(inputStream);
075      assertNotNull(data);
076      assertEquals(1, data.size());
077      assertTrue(data.containsKey("units"));
078      @SuppressWarnings("unchecked")
079      final Collection<Map<String, Object>> units = (Collection<Map<String, Object>>)data.get("units");
080      assertNotNull(units);
081      assertEquals(2, units.size());
082      for (final Map<String, Object> unit : units) {
083        assertNotNull(unit);
084        assertTrue(unit.containsKey("name"));
085      }
086    }
087  }
088
089  @Test
090  public void testUrlNormalization() throws IOException {
091    final URL url = new URL("file:/foo/META-INF/persistence.xml");
092    final URL relative = new URL(url, "..");
093    assertEquals("file:/foo/", relative.toString());
094  }
