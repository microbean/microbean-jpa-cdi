072/**
073 * A {@linkplain Extension portable extension} normally instantiated
074 * by the Java {@linkplain ServiceLoader service provider
075 * infrastructure} that integrates the provider-independent parts of
076 * <a href="https://javaee.github.io/tutorial/partpersist.html#BNBPY"
077 * target="_parent">JPA</a> into CDI.
078 *
079 * @author <a href="https://about.me/lairdnelson"
080 * target="_parent">Laird Nelson</a>
081 *
082 * @see PersistenceUnitInfoBean
083 */
084public class JpaExtension implements Extension {

100  /**
101   * A {@link Map} of {@link Set}s of {@link Class}es whose keys are
102   * persistence unit names and whose values are {@link Set}s of
103   * {@link Class}es discovered by CDI (and hence consist of unlisted
104   * classes in the sense that they might not be found in any {@link
105   * PersistenceUnitInfo}).
106   *
107   * <p>Such {@link Class}es, of course, might not have been weaved
108   * appropriately by the relevant {@link PersistenceProvider}.</p>
109   *
110   * <p>This field is never {@code null}.</p>
111   */
112  private final Map<String, Set<Class<?>>> unlistedManagedClassesByPersistenceUnitNames;
113

116   * Constructors.
117   */

120  /**
121   * Creates a new {@link JpaExtension}.
122   */
123  public JpaExtension() {
124    super();
125    this.unlistedManagedClassesByPersistenceUnitNames = new HashMap<>();
126  }

134  private final void discoverManagedClasses(@Observes
135                                            @WithAnnotations({
136                                              Converter.class,
137                                              Entity.class,
138                                              Embeddable.class,
139                                              MappedSuperclass.class
140                                            })
141                                            final ProcessAnnotatedType<?> event) {
142    if (event != null) {
143      final AnnotatedType<?> annotatedType = event.getAnnotatedType();
144      if (annotatedType != null) {
145        final Class<?> managedClass = annotatedType.getJavaClass();
146        assert managedClass != null;
147        final Set<PersistenceUnit> persistenceUnits = annotatedType.getAnnotations(PersistenceUnit.class);
148        if (persistenceUnits == null || persistenceUnits.isEmpty()) {
149          Set<Class<?>> unlistedManagedClasses = this.unlistedManagedClassesByPersistenceUnitNames.get("");
150          if (unlistedManagedClasses == null) {
151            unlistedManagedClasses = new HashSet<>();
152            this.unlistedManagedClassesByPersistenceUnitNames.put("", unlistedManagedClasses);
153          }
154          unlistedManagedClasses.add(managedClass);
155        } else {
156          for (final PersistenceUnit persistenceUnit : persistenceUnits) {
157            String name = "";
158            if (persistenceUnit != null) {
159              name = persistenceUnit.unitName();
160              assert name != null;
161            }
162            Set<Class<?>> unlistedManagedClasses = this.unlistedManagedClassesByPersistenceUnitNames.get(name);
163            if (unlistedManagedClasses == null) {
164              unlistedManagedClasses = new HashSet<>();
165              this.unlistedManagedClassesByPersistenceUnitNames.put(name, unlistedManagedClasses);
166            }
167            unlistedManagedClasses.add(managedClass);
168          }
169        }
170      }
171      event.veto(); // managed classes can't be beans
172    }
173  }

175  private final void afterBeanDiscovery(@Observes final AfterBeanDiscovery event, final BeanManager beanManager)
176    throws IOException, JAXBException, ReflectiveOperationException {
177    if (event != null && beanManager != null) {
178
179      // Add a bean for PersistenceProviderResolver.
180      final PersistenceProviderResolver resolver =
181        PersistenceProviderResolverHolder.getPersistenceProviderResolver();
182      event.addBean()
183        .types(PersistenceProviderResolver.class)
184        .scope(Singleton.class)
185        .createWith(cc -> resolver);
186
187      // Add a bean for each "generic" PersistenceProvider reachable
188      // from the resolver.  (Any PersistenceUnitInfo may also specify
189      // the class name of a PersistenceProvider whose class may not
190      // be among those loaded by the resolver; we deal with those
191      // later.)
192      final Collection<? extends PersistenceProvider> providers = resolver.getPersistenceProviders();
193      for (final PersistenceProvider provider : providers) {
194        event.addBean()
195          .addTransitiveTypeClosure(provider.getClass())
196          .scope(Singleton.class)
197          .createWith(cc -> provider);
198      }
199
200      // Collect all pre-existing PersistenceUnitInfo beans and make
201      // sure their associated PersistenceProviders are beanified.
202      // (Many times this Set will be empty.)
203      final Set<Bean<?>> preexistingPersistenceUnitInfoBeans = beanManager.getBeans(PersistenceUnitInfo.class);
204      if (preexistingPersistenceUnitInfoBeans != null && !preexistingPersistenceUnitInfoBeans.isEmpty()) {
205        for (final Bean<?> preexistingPersistenceUnitInfoBean : preexistingPersistenceUnitInfoBeans) {
206          if (preexistingPersistenceUnitInfoBean != null) {
207            // We use the Bean directly to create a
208            // PersistenceUnitInfo instance.  This instance is by
209            // definition unmanaged by CDI, which is fine in this
210            // narrow case: we throw it away immediately.  We need it
211            // only for the return values of its
212            // getPersistenceProviderClassName() and getClassLoader()
213            // methods.
214            final Object pui = preexistingPersistenceUnitInfoBean.create(null);
215            if (pui instanceof PersistenceUnitInfo) {
216              maybeAddPersistenceProviderBean(event, (PersistenceUnitInfo)pui, providers);
217            }
218          }
219        }
220      }
221      
222      // Discover all META-INF/persistence.xml resources, load them
223      // using JAXB, and turn them into PersistenceUnitInfo instances,
224      // and add beans for all of them as well as their associated
225      // PersistenceProviders (if applicable).
226      final ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
227      assert classLoader != null;
228      final Enumeration<URL> urls = classLoader.getResources("META-INF/persistence.xml");
229      if (urls != null && urls.hasMoreElements()) {
230        final Supplier<? extends ClassLoader> tempClassLoaderSupplier;
231        if (classLoader instanceof URLClassLoader) {
232          tempClassLoaderSupplier = () -> new URLClassLoader(((URLClassLoader)classLoader).getURLs());
233        } else {
234          tempClassLoaderSupplier = () -> classLoader;
235        }
236        final Unmarshaller unmarshaller =
237          JAXBContext.newInstance(JAXB_GENERATED_PACKAGE_NAME).createUnmarshaller();
238        assert unmarshaller != null;
239        while (urls.hasMoreElements()) {
240          final URL url = urls.nextElement();
241          final Collection<? extends PersistenceUnitInfo> persistenceUnitInfos =
242            PersistenceUnitInfoBean.fromPersistence((Persistence)unmarshaller.unmarshal(url),
243                                                    classLoader,
244                                                    tempClassLoaderSupplier,
245                                                    new URL(url, ".."), // e.g. META-INF/..
246                                                    this.unlistedManagedClassesByPersistenceUnitNames,
247                                                    (jta, useDefaultJta, dataSourceName) -> getDataSource(jta, useDefaultJta, dataSourceName, beanManager));
248          for (final PersistenceUnitInfo persistenceUnitInfo : persistenceUnitInfos) {
249            assert persistenceUnitInfo != null;
250
251            String persistenceUnitName = persistenceUnitInfo.getPersistenceUnitName();
252            if (persistenceUnitName == null) {
253              persistenceUnitName = "";
254            }
255
256            event.addBean()
257              .types(Collections.singleton(PersistenceUnitInfo.class))
258              .scope(Singleton.class)
259              .addQualifiers(NamedLiteral.of(persistenceUnitName))
260              .createWith(cc -> persistenceUnitInfo);
261
262            maybeAddPersistenceProviderBean(event, persistenceUnitInfo, providers);
263            
264          }
265        }
266      }
267    }
268  }

270  private static final void maybeAddPersistenceProviderBean(final AfterBeanDiscovery event,
271                                                            final PersistenceUnitInfo persistenceUnitInfo,
272                                                            final Collection<? extends PersistenceProvider> providers)
273    throws ReflectiveOperationException {
274    Objects.requireNonNull(event);
275    Objects.requireNonNull(persistenceUnitInfo);
276    final String providerClassName = persistenceUnitInfo.getPersistenceProviderClassName();
277    if (providerClassName != null) {
278
279      boolean add = true;
280      
281      if (providers != null && !providers.isEmpty()) {
282        for (final PersistenceProvider provider : providers) {
283          if (provider != null && provider.getClass().getName().equals(providerClassName)) {
284            add = false;
285            break;
286          }
287        }
288      }
289
290      if (add) {
291      
292        // The PersistenceProvider class in question is not one we
293        // already loaded.  Add a bean for it too.
294        event.addBean()
295          .types(PersistenceProvider.class)
296          .scope(Singleton.class)
297          .createWith(cc -> {
298              try {
299                ClassLoader classLoader = persistenceUnitInfo.getClassLoader();
300                if (classLoader == null) {
301                  classLoader = Thread.currentThread().getContextClassLoader();
302                }
303                assert classLoader != null;
304                @SuppressWarnings("unchecked")
305                final Class<? extends PersistenceProvider> c = (Class<? extends PersistenceProvider>)Class.forName(providerClassName, true, classLoader);
306                return c.newInstance();
307              } catch (final ReflectiveOperationException reflectiveOperationException) {
308                throw new CreationException(reflectiveOperationException.getMessage(), reflectiveOperationException);
309              }
310            });
311      }
312      
313    }
314  }

316  private static final DataSource getDataSource(final boolean jta, final boolean useDefaultJta, final String dataSourceName, final BeanManager beanManager) {
317    Objects.requireNonNull(beanManager);
318    final Bean<?> bean;
319    if (jta) {
320      if (useDefaultJta) {
321        // Ignore, on purpose, the supplied dataSourceName.
322        bean = beanManager.resolve(beanManager.getBeans(DataSource.class));
323      } else if (dataSourceName == null) {
324        bean = null;
325      } else {
326        bean = beanManager.resolve(beanManager.getBeans(DataSource.class, NamedLiteral.of(dataSourceName)));
327      }
328    } else if (dataSourceName == null) {
329      bean = null;
330    } else {
331      bean = beanManager.resolve(beanManager.getBeans(DataSource.class, NamedLiteral.of(dataSourceName)));
332    }
333    final DataSource returnValue;
334    if (bean == null) {
335      returnValue = null;
336    } else {
337      returnValue = (DataSource)beanManager.getReference(bean, DataSource.class, beanManager.createCreationalContext(bean));
338    }
339    return returnValue;
340  }
